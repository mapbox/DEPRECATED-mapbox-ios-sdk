--- category: examples layout: examples title: Drawing a MapBox marker description: Creating a default marker with a callout and title at a point. tags: - marker - point - pin - callout - popup --- - (void)viewDidLoad { [super viewDidLoad]; RMMapBoxSource *tileSource = [[RMMapBoxSource alloc] initWithMapID:@"examples.map-z2effxa8"]; RMMapView *mapView = [[RMMapView alloc] initWithFrame:self.view.bounds andTilesource:tileSource]; [self.view addSubview:mapView]; RMPointAnnotation *annotation = [[RMPointAnnotation alloc] initWithMapView:mapView coordinate:mapView.centerCoordinate andTitle:@"Hello, world!"]; [mapView addAnnotation:annotation]; }
